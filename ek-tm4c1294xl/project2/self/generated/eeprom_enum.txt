
typedef enum
{
  EEPROM_BEGIN          = 0,

  EEPROM_LABEL          = EEPROM_BEGIN,

  EEPROM_IP             = EEPROM_LABEL + 4,
  EEPROM_GATEWAY        = EEPROM_IP + 4,
  EEPROM_NETMASK        = EEPROM_GATEWAY + 4,
  EEPROM_PORT           = EEPROM_NETMASK + 4,

  EEPROM_DEVICE_NAME    = EEPROM_PORT + 4,
  EEPROM_OWNER_NAME     = EEPROM_DEVICE_NAME + 4*3,

  EEPROM_CONNECTION_TIMEOUT = EEPROM_OWNER_NAME + 4*3,
  EEPROM_ROUTING_MODE   = EEPROM_CONNECTION_TIMEOUT + 4,
  EEPROM_DEST_IP        = EEPROM_ROUTING_MODE + 4,
  EEPROM_DEST_PORT      = EEPROM_DEST_IP + 4,

  EEPROM_BAUD_RATE_0    = EEPROM_DEST_PORT + 4,
  EEPROM_BAUD_RATE_1    = EEPROM_BAUD_RATE_0 + 4,

  EEPROM_UDP_DEBUG_FLAG = EEPROM_BAUD_RATE_1 + 4,
  EEPROM_UDP_DEBUG_PORT = EEPROM_UDP_DEBUG_FLAG + 4,

  EEPROM_END            = EEPROM_UDP_DEBUG_PORT + 4
} eeprom;
